Single node
#Install java and set path
sudo apt install openjdk-11-jdk -y
java -version
nano ~/.bashrc

#in ~/.bashrc
# add these two lines
export JAVA_HOME=/usr/lib/jvm/java-11-openjdk-amd64
export PATH=$PATH:$JAVA_HOME/bin

source ~/.bashrc

#Install SSH and enable it
sudo apt install openssh-server -y
sudo systemctl start ssh
sudo systemctl enable ssh
ssh localhost
#it will ask for permission so type "yes" then give password as hadoop ( that is given for sudo)
ssh-keygen -t rsa
#press enter for all prompts
cat ~/.ssh/id_rsa.pub >> ~/.ssh/authorized_keys
chmod 600 ~/.ssh/authorized_keys
ssh localhost
#now ssh localhost command works without asking any password

#Download Hadoop and extract it
wget https://downloads.apache.org/hadoop/common/hadoop-3.3.6/hadoop-3.3.6.tar.gz
tar -xvzf hadoop-3.3.6.tar.gz
sudo mv hadoop-3.3.6 /usr/local/hadoop

sudo chown -R $USER:$USER /usr/local/hadoop
nano ~/.bashrc
# in this add 
export HADOOP_HOME=/usr/local/hadoop
export PATH=$PATH:$HADOOP_HOME/bin:$HADOOP_HOME/sbin
source ~/.bashrc

nano $HADOOP_HOME/etc/hadoop/hadoop-env.sh

#in this search for export JAVA_HOME by using clt+w
#and uncomment that line and make it as 
export JAVA_HOME=/usr/lib/jvm/java-11-openjdk-amd64

cd $HADOOP_HOME/etc/hadoop
nano core-site.xml
#in this replace all content with below code 
<configuration>
  <property>
    <name>fs.defaultFS</name>
    <value>hdfs://localhost:9000</value>
  </property>
</configuration>

nano hdfs-site.xml
#in this replace all content with code below
<configuration>
  <property>
    <name>dfs.replication</name>
    <value>1</value>
  </property>

  <property>
    <name>dfs.namenode.name.dir</name>
    <value>file:/usr/local/hadoop/data/namenode</value>
  </property>

  <property>
    <name>dfs.datanode.data.dir</name>
    <value>file:/usr/local/hadoop/data/datanode</value>
  </property>
</configuration>
cd  -- must to come out of directory

mkdir -p /usr/local/hadoop/data/namenode

mkdir -p /usr/local/hadoop/data/datanode

hdfs namenode -format

start-dfs.sh

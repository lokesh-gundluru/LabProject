# This program builds and trains a simple LSTM model for sentiment analysis on the IMDB movie review dataset. It loads the dataset, preprocesses it by padding sequences, constructs the LSTM model, compiles it, and trains it on the training data. After training, it evaluates the model's accuracy on the test set and visualizes the training history. Finally, it makes sample predictions on a few test reviews and compares them to the actual labels.
# Note: Ensure you have TensorFlow and Matplotlib installed to run this code.
import numpy as np
import matplotlib.pyplot as plt
from tensorflow.keras.datasets import imdb
from tensorflow.keras.preprocessing.sequence import pad_sequences
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Embedding, LSTM, Dense

# === LOAD DATA ===
vocab_size = 10000
max_len = 200

(X_train, y_train), (X_test, y_test) = imdb.load_data(num_words=vocab_size)

X_train = pad_sequences(X_train, maxlen=max_len)
X_test = pad_sequences(X_test, maxlen=max_len)

# === BUILD MODEL ===
model = Sequential([
    Embedding(vocab_size, 128, input_length=max_len),
    LSTM(64),
    Dense(1, activation='sigmoid')
])

# === COMPILE ===
model.compile(optimizer='adam',
              loss='binary_crossentropy',
              metrics=['accuracy'])

# === TRAIN ===
history = model.fit(X_train, y_train,
                    epochs=3,
                    batch_size=64,
                    validation_split=0.2)

# === EVALUATE ===
acc = model.evaluate(X_test, y_test, verbose=0)[1]
print("Test Accuracy:", acc)

# === PLOT ===
plt.plot(history.history['accuracy'], label='Train')
plt.plot(history.history['val_accuracy'], label='Validation')
plt.title("Accuracy")
plt.legend()
plt.show()

# === SAMPLE PREDICTIONS ===
pred = (model.predict(X_test[:5]) > 0.5).astype(int)

for i in range(5):
    print("Review", i+1, "Prediction:", pred[i][0], "Actual:", y_test[i])

⚙️ PART 2: Prerequisites
🔹 Update system
sudo apt update && sudo apt upgrade -y
☕ PART 3: JAVA 8 INSTALLATION
✅ OPTION 1: APT METHOD (if available)
sudo apt install openjdk-8-jdk -y

Check:

java -version

👉 If this fails → use manual method below

✅ OPTION 2: MANUAL METHOD (RECOMMENDED)
🔹 Download Java 8
wget https://github.com/adoptium/temurin8-binaries/releases/latest/download/OpenJDK8U-jdk_x64_linux_hotspot.tar.gz
🔹 Extract
tar -xvzf OpenJDK8U-jdk_x64_linux_hotspot.tar.gz
🔹 Move
sudo mv jdk8u* /usr/local/java8
🔹 Set JAVA_HOME
nano ~/.bashrc

Add:

export JAVA_HOME=/usr/local/java8
export PATH=$PATH:$JAVA_HOME/bin
🔹 Apply
source ~/.bashrc
🔹 Verify
java -version
javac -version
🔐 PART 4: SSH SETUP
sudo apt install openssh-server -y
sudo systemctl start ssh
sudo systemctl enable ssh
🔹 Passwordless SSH
ssh-keygen -t rsa
cat ~/.ssh/id_rsa.pub >> ~/.ssh/authorized_keys
chmod 600 ~/.ssh/authorized_keys
ssh localhost
📦 PART 5: Hadoop Installation
🔹 Download
wget https://downloads.apache.org/hadoop/common/hadoop-3.3.6/hadoop-3.3.6.tar.gz
🔹 Extract & Move
tar -xvzf hadoop-3.3.6.tar.gz
sudo mv hadoop-3.3.6 /usr/local/hadoop
sudo chown -R $USER:$USER /usr/local/hadoop
🔹 Set Variables
nano ~/.bashrc

Add:

export HADOOP_HOME=/usr/local/hadoop
export PATH=$PATH:$HADOOP_HOME/bin:$HADOOP_HOME/sbin
export JAVA_HOME=/usr/local/java8

Apply:

source ~/.bashrc
⚙️ PART 6: Hadoop Configuration
🔹 hadoop-env.sh
nano $HADOOP_HOME/etc/hadoop/hadoop-env.sh

Set:

export JAVA_HOME=/usr/local/java8
🔹 core-site.xml
cd $HADOOP_HOME/etc/hadoop
nano core-site.xml
<configuration>
  <property>
    <name>fs.defaultFS</name>
    <value>hdfs://localhost:9000</value>
  </property>
</configuration>
🔹 hdfs-site.xml
nano hdfs-site.xml
<configuration>
  <property>
    <name>dfs.replication</name>
    <value>1</value>
  </property>
  <property>
    <name>dfs.namenode.name.dir</name>
    <value>file:/usr/local/hadoop/data/namenode</value>
  </property>
  <property>
    <name>dfs.datanode.data.dir</name>
    <value>file:/usr/local/hadoop/data/datanode</value>
  </property>
</configuration>
🔹 Create directories
mkdir -p /usr/local/hadoop/data/namenode
mkdir -p /usr/local/hadoop/data/datanode
🚀 PART 7: Run Hadoop
🔹 Format
hdfs namenode -format
🔹 Start
start-dfs.sh
🔹 Check
jps
🔹 Verify
hdfs dfs -ls /
🌐 Web UI
http://localhost:9870

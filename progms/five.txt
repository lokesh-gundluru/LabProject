# This program compares the performance of GRU and LSTM models on the IMDB movie review dataset for sentiment analysis. It loads the dataset, preprocesses it, trains both models, evaluates their accuracy, and visualizes the results using a bar graph.
# Note: Ensure you have TensorFlow and Matplotlib installed to run this code.
import numpy as np
import matplotlib.pyplot as plt
from tensorflow.keras.datasets import imdb
from tensorflow.keras.preprocessing.sequence import pad_sequences
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Embedding, GRU, LSTM, Dense

# === LOAD DATA ===
vocab_size = 10000
max_len = 200

(X_train, y_train), (X_test, y_test) = imdb.load_data(num_words=vocab_size)

X_train = pad_sequences(X_train, maxlen=max_len)
X_test = pad_sequences(X_test, maxlen=max_len)

# === GRU MODEL ===
gru_model = Sequential([
    Embedding(vocab_size, 128, input_length=max_len),
    GRU(64),
    Dense(1, activation='sigmoid')
])

gru_model.compile(optimizer='adam',
                  loss='binary_crossentropy',
                  metrics=['accuracy'])

print("Training GRU Model...")
gru_model.fit(X_train, y_train, epochs=3, batch_size=64, validation_split=0.2)

gru_acc = gru_model.evaluate(X_test, y_test, verbose=0)[1]

# === LSTM MODEL ===
lstm_model = Sequential([
    Embedding(vocab_size, 128, input_length=max_len),
    LSTM(64),
    Dense(1, activation='sigmoid')
])

lstm_model.compile(optimizer='adam',
                   loss='binary_crossentropy',
                   metrics=['accuracy'])

print("Training LSTM Model...")
lstm_model.fit(X_train, y_train, epochs=3, batch_size=64, validation_split=0.2)

lstm_acc = lstm_model.evaluate(X_test, y_test, verbose=0)[1]

# === RESULTS ===
print("GRU Accuracy :", gru_acc)
print("LSTM Accuracy:", lstm_acc)

# === BAR GRAPH ===
models = ['GRU', 'LSTM']
acc = [gru_acc, lstm_acc]

plt.bar(models, acc)
plt.title("GRU vs LSTM Accuracy")
plt.ylabel("Accuracy")
plt.show()
